Script Python para Geração com ReportLab
python



# reports/pdf_generator.py
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer, Table, Image, PageBreak
from reportlab.lib.styles import getSampleStyleSheet, ParagraphStyle
from reportlab.lib import colors
from reportlab.lib.pagesizes import A4
from reportlab.lib.units import cm
import qrcode  # para gerar QR code

def generate_student_report_pdf(student_data: dict, output_path: str):
    doc = SimpleDocTemplate(output_path, pagesize=A4, 
                           rightMargin=2*cm, leftMargin=2*cm,
                           topMargin=2*cm, bottomMargin=2*cm)
    styles = getSampleStyleSheet()
    story = []
    
    # Estilo personalizado para títulos
    title_style = ParagraphStyle('CustomTitle', parent=styles['Heading1'], 
                                fontSize=16, spaceAfter=12, textColor=colors.HexColor('#2563EB'))
    
    # Cabeçalho
    story.append(Paragraph(f"Relatório Pedagógico - {student_data['name']}", title_style))
    story.append(Paragraph(f"Turma: {student_data['class']} | Data: {datetime.now().strftime('%d/%m/%Y')}", styles['Normal']))
    story.append(Spacer(1, 0.5*cm))
    
    # Métricas principais em tabela
    metrics_data = [
        ['Média Geral', f"{student_data['overall_score']}%"],
        ['Questões Consolidadas', f"{student_data['consolidated_count']}/450"],
        ['Streak Atual', f"🔥 {student_data['current_streak']}"],
        ['Percentil na Turma', f"{student_data['percentile']}º"]
    ]
    metrics_table = Table(metrics_data, colWidths=[4*cm, 3*cm])
    metrics_table.setStyle([
        ('BACKGROUND', (0, 0), (-1, 0), colors.HexColor('#EFF6FF')),
        ('GRID', (0, 0), (-1, -1), 0.5, colors.grey),
        ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'),
        ('ALIGN', (1, 0), (1, -1), 'RIGHT')
    ])
    story.append(metrics_table)
    story.append(Spacer(1, 1*cm))
    
    # Heatmap de conceitos (simulado com tabela colorida)
    story.append(Paragraph("🔥 Conceitos para Reforçar", styles['Heading2']))
    heatmap_data = [['Conceito', 'Erros', 'Prioridade']]
    for concept in student_data['weak_concepts'][:5]:
        priority_color = colors.red if concept['priority'] == 'high' else colors.orange
        heatmap_data.append([
            concept['label'], 
            str(concept['error_count']), 
            Paragraph(f"<font color='{priority_color.hexval()}'>●</font> {concept['priority']}")
        ])
    heatmap_table = Table(heatmap_data, colWidths=[5*cm, 2*cm, 2*cm])
    heatmap_table.setStyle([
        ('BACKGROUND', (0, 0), (-1, 0), colors.lightgrey),
        ('GRID', (0, 0), (-1, -1), 0.5, colors.grey),
        ('VALIGN', (0, 0), (-1, -1), 'MIDDLE')
    ])
    story.append(heatmap_table)
    story.append(Spacer(1, 1*cm))
    
    # Plano de intervenção
    story.append(Paragraph("🎯 Plano de Ação Personalizado", styles['Heading2']))
    for intervention in student_data['intervention_plan']:
        story.append(Paragraph(f"• <b>{intervention['concept']}</b>: {intervention['activity']}", styles['Normal']))
        story.append(Paragraph(f"  <i>Critério: {intervention['success_criteria']}</i>", 
                            ParagraphStyle('Italic', parent=styles['Normal'], fontSize=9, textColor=colors.grey)))
    
    story.append(PageBreak())
    
    # QR Code para acesso digital
    qr = qrcode.QRCode(version=1, box_size=10, border=4)
    qr.add_data(f"https://eduquest.app/report/{student_data['student_id']}")
    qr.make(fit=True)
    img = qr.make_image(fill_color="black", back_color="white")
    img_path = f"/tmp/qr_{student_data['student_id']}.png"
    img.save(img_path)
    
    story.append(Paragraph("📱 Acesse seu relatório interativo:", styles['Normal']))
    story.append(Image(img_path, width=3*cm, height=3*cm))
    story.append(Paragraph("<i>Escaneie com a câmera do celular</i>", 
                          ParagraphStyle('Caption', parent=styles['Normal'], fontSize=8, alignment=1)))
    
    # Footer com LGPD
    story.append(Spacer(1, 2*cm))
    footer_style = ParagraphStyle('Footer', parent=styles['Normal'], fontSize=7, textColor=colors.grey, alignment=1)
    story.append(Paragraph("Relatório gerado automaticamente • Dados protegidos conforme LGPD • Eduquest © 2026", footer_style))
    
    doc.build(story)
